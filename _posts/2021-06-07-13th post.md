  const {type,payload}=action;

      return {...state,loading:true};
    }
    case 'SUCCESS':{
      return {...state, payload};
    }
    case 'FAILURE':{
      return {...state,error:true};

      return {...state,loading:false};
    }
    default:
      state;
  }
}
