#include<iostream>
#include<sstream>
using namespace std;

int circular[10001];
int cnt;

bool isPalindrome(int x) {
	stringstream ss;
	ss << x;
	string str = ss.str();
	int len = str.length();
	for (int i = 0; i < len; i++) {
		if (str[i] != str[len - 1 - i])
			return false;
		return true;
	}

bool one(int x) {
	for (int i = 0; i < cnt; i++) {
		if (circular[i] == x) {
			cout << 1 << " " << circular[i] << "\n";
			return true;
		}
	}
	return false;
}

bool two(int x) {
	for (int i = 0; i < cnt; i++) {
		for (int j = 0; j < cnt; j++) {
			if (circular[i] + circular[j] == x) {
				cout << 2 << " " << circular[j] <<" "<< circular[i] << "\n";
				return true;
			}
		}
	}
	return false;
}

bool three(int x) {
	for (int i = 0; i < cnt; i++) {
		for (int j = 0; j < cnt; j++) {
			for (int k = 0; k < cnt; k++) {
				if (circular[i] + circular[j]+circular[k] == x) {
					cout << 2 << " " <<circular[k]<<" "<< circular[j] << " " << circular[i] << "\n";
					return true;
				}
			}
		}
	}
	return false;
}

	int T, testCase;

		if (isPalindrome(i)) {

		}

		}
	}
}
